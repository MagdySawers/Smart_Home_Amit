
#include <stdio.h>
#include <stdlib.h>
#include <string.h>


struct student
{
    char name[20];
    int age;
    int degree;
    int section;
}; 

void print (struct student * p)
{
 
for(int i=0;i<6;i++)
  {

    printf("Student  %d\n",i+1);
    printf("\n");
    printf("Name   : %s\n",p[i].name);
    printf("Age    : %d\n",p[i].age);
    printf("Degree : %d\n",p[i].degree);
    printf("Section: %d\n",p[i].section);
    printf("\n");

   }
}
//----------------------------------------------

int main()
{
    struct student s[5];

for(int i=0;i<6;i++)
  {

    printf("Student  %d\n",i+1);
    printf("Name   : ");
    scanf("%s",&s[i].name);
    printf("Age    : ");
    scanf("%d",&s[i].age);
    printf("Degree : ");
    scanf("%d",&s[i].degree);
    printf("Section: ");
    scanf("%d",&s[i].section);
    printf("\n");

  }


print(&s);


    return 0;
}
